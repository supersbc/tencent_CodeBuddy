### 3.3 核心算法

#### 3.3.1 规模判断算法
```python
def determine_scale(qps, data_volume, concurrent_users):
    """
    基于多维度参数判断系统规模
    """
    score = 0
    
    # QPS评分
    if qps < 1000: score += 1
    elif qps < 5000: score += 2
    elif qps < 10000: score += 3
    else: score += 4
    
    # 数据量评分
    if data_volume < 100: score += 1
    elif data_volume < 500: score += 2
    elif data_volume < 2000: score += 3
    else: score += 4
    
    # 并发用户评分
    if concurrent_users < 1000: score += 1
    elif concurrent_users < 10000: score += 2
    elif concurrent_users < 50000: score += 3
    else: score += 4
    
    # 规模判定
    if score <= 4: return 'SMALL'
    elif score <= 7: return 'MEDIUM'
    elif score <= 10: return 'LARGE'
    else: return 'XLARGE'
```

#### 3.3.2 分片计算算法
```python
def calculate_shards(data_volume, qps):
    """
    根据数据量和QPS计算最优分片数
    """
    # 基于数据量的分片数（每个分片最大500GB）
    data_shards = math.ceil(data_volume / 500)
    
    # 基于QPS的分片数（每个分片最大5000 QPS）
    qps_shards = math.ceil(qps / 5000)
    
    # 取较大值，确保满足两个维度的需求
    shard_count = max(data_shards, qps_shards)
    
    # 分片数调整为2的幂次（便于扩展）
    return 2 ** math.ceil(math.log2(shard_count))
```

#### 3.3.3 副本数计算算法
```python
def calculate_replicas(ha_level, industry):
    """
    根据高可用等级和行业特性计算副本数
    """
    base_replicas = {
        'low': 1,      # 单副本
        'medium': 2,   # 双副本
        'high': 3      # 三副本
    }
    
    # 金融行业额外增加副本
    if industry == '金融' and ha_level == 'high':
        return 4
    
    return base_replicas.get(ha_level, 2)
```

#### 3.3.4 成本计算算法
```python
def calculate_cost(equipment_list):
    """
    计算总拥有成本（TCO）
    """
    # 硬件成本
    hardware_cost = sum([item['price'] * item['quantity'] 
                        for item in equipment_list])
    
    # 软件成本（TDSQL许可证：硬件成本的15%）
    software_cost = hardware_cost * 0.15
    
    # 实施服务成本（硬件成本的10%）
    service_cost = hardware_cost * 0.10
    
    # 年度运营成本
    annual_operation = {
        'power': hardware_cost * 0.05,      # 电费
        'bandwidth': 120000,                 # 带宽费用
        'maintenance': hardware_cost * 0.08, # 维护费用
        'personnel': 500000                  # 人员成本
    }
    
    # 初始投资
    initial_investment = hardware_cost + software_cost + service_cost
    
    # 3年TCO
    three_year_tco = initial_investment + sum(annual_operation.values()) * 3
    
    return {
        'initial_investment': initial_investment,
        'annual_operation': sum(annual_operation.values()),
        'three_year_tco': three_year_tco
    }
```
